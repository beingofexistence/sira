## 尺寸

    <input defaultChecked className="radio colored warn md" type="radio" />
    <input defaultChecked className="radio colored warn lg" type="radio" />
    <input defaultChecked className="radio colored warn xl" type="radio" />
  </div>
</CodeDemo>
